# 栈

```c++
const int N = 100010;

int stk[N], tt;

void init(){
	tt = 0;
}

void push(int x){
	stk[ ++ tt] = x;
}

void pop(int x){
	tt -- ;
}

bool empty(){
	if (tt > 0)
		return 0;
	else 
		return 1;	
}
```

# 队列

```c++
const int N = 100010;

int q[N], hh, tt = -1;

void push(int x){
	q[++ tt] = x;
}

void pop(){
	hh ++;
}

bool empty(){
	if(hh <= tt)
		return 0;
	else
		return 1;
}
```
