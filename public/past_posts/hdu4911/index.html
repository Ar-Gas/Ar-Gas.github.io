#include <iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>

using namespace std;

const int N = 100005;

int n;
long long k,ans;
int a[N],b[N];


void merge(int l, int mid, int r){
    int i = l, j = mid + 1;
    int idx = 0;
    while(i <= mid && j <= r){
        if(a[i] > a[j]){
            b[idx ++] = a[j ++];
            ans += mid - i + 1;
        }
        else b[idx ++] = a[i ++];
    }
    while(i <= mid )b[idx ++] = a[i ++];
    while(j <= r )b[idx ++] = a[j ++];
    for (int i = 0; i < idx; ++i)
    {
        a[l + i] = b[i];
    }
    return;
}

void merge_sort(int l, int r){
    int mid = (l + r) / 2;
    if(l >= r)return;
    else{
        merge_sort(l, mid);
        merge_sort(mid +1, r);
        merge(l, mid, r);
    }
    return;
}


int main()
{
    ios::sync_with_stdio(false),cin.tie(0),cout.tie(0);
    while(cin >> n >> k){
        memset(b, 0, N);
        memset(a, 0, N);
        ans = 0;
        for (int i = 0; i < n; ++i)
        {
            cin >> a[i];
        }
        merge_sort(0, n - 1);
        cout << (ans - k > 0 ? (long long)(ans - k) : 0) << endl;
    }
    return 0;
}
